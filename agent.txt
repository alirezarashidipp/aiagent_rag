import yaml
from pathlib import Path
from models import CoreElements, ValidationResult, AcceptanceCriteria, MissingTechnicalDetail, FinalStory
from llm_client import StructuredLLMClient

class JiraGroomingAgent:
    def __init__(self, llm_client: StructuredLLMClient, prompt_file: str = "prompts.yaml"):
        self.llm = llm_client
        self.state = CoreElements(who=None, what=None, why=None)
        self.ac = []
        self.prompts = self._load_prompts(prompt_file)

    def _load_prompts(self, filepath: str) -> dict:
        path = Path(filepath)
        if not path.exists():
            raise FileNotFoundError(f"[FATAL] Prompt configuration file missing: {filepath}")
        with open(path, "r", encoding="utf-8") as file:
            return yaml.safe_load(file)

    def run(self):
        print("--- AI Jira Grooming Assistant ---")
        raw_input = input("1. Please provide the raw Jira description:\n> ")
        
        self._extract_core_elements(raw_input)
        
        # Validates core requirements in order: WHAT, WHO, WHY
        self._clarify_missing_context()
        
        # Finalize technical constraints before drafting criteria
        self._refine_technical_details()
        
        # AC generation utilizes the fully refined 'WHAT' statement
        self._generate_and_review_ac()
        
        self._generate_final_story()

    def _extract_core_elements(self, raw_input: str):
        sys_prompt = self.prompts['system_personas']['analyst']
        user_prompt = self.prompts['user_prompts']['extract_core'].format(raw_input=raw_input)
        
        self.state = self.llm.query(sys_prompt, user_prompt, CoreElements)

    def _clarify_missing_context(self):
        sys_prompt = self.prompts['system_personas']['validator']
        max_retries = 3  # Production circuit breaker
        
        # 'what' is now strictly validated first, preventing silent failures on generic initial inputs
        for field in ['what', 'who', 'why']:
            retries = 0
            last_user_ans = None
            
            while getattr(self.state, field) is None:
                if retries >= max_retries:
                    print(f"\n[SYSTEM WARN] Validation threshold exceeded. Forcing ingestion of raw input.")
                    setattr(self.state, field, last_user_ans or f"Unknown {field}")
                    break
                    
                user_ans = input(f"\nMissing {field.upper()}. Please clarify:\n> ").strip()
                if not user_ans:
                    print("Input cannot be empty. Try again.")
                    continue
                    
                last_user_ans = user_ans
                
                user_prompt = self.prompts['user_prompts']['validate_missing'].format(
                    field_name=field.upper(),
                    user_ans=user_ans
                )
                
                validation = self.llm.query(sys_prompt, user_prompt, ValidationResult)
                
                if validation.is_valid:
                    setattr(self.state, field, validation.normalized_value)
                else:
                    retries += 1
                    print(f"The LLM rejected the input for lacking detail. (Attempt {retries}/{max_retries})")

    def _refine_technical_details(self):
        sys_prompt = self.prompts['system_personas']['tech_lead']
        user_prompt = self.prompts['user_prompts']['refine_tech'].format(
            what_statement=self.state.what
        )
        
        analysis = self.llm.query(sys_prompt, user_prompt, MissingTechnicalDetail)
        
        if analysis.has_missing_details and analysis.questions:
            print(f"\n[SYSTEM] The technical lead identified {len(analysis.questions)} potential missing detail(s).")
            
            for i, question in enumerate(analysis.questions, 1):
                ans = input(f"\nDetail {i}/{len(analysis.questions)}:\n{question}\n(Answer, or type 'skip' / 'not important'):\n> ").strip()
                
                # Check against common dismissal phrases
                if ans.lower() not in ["not important", "skip", "no", "n/a", ""]:
                    self.state.what += f" [Technical Context: {ans}]"

    def _generate_and_review_ac(self):
        print("\nGenerating Acceptance Criteria...")
        sys_prompt = self.prompts['system_personas']['agile_coach']
        user_prompt = self.prompts['user_prompts']['generate_ac'].format(
            state_json=self.state.model_dump_json()
        )
        
        ac_draft = self.llm.query(sys_prompt, user_prompt, AcceptanceCriteria)
        
        print("\nProposed Acceptance Criteria:")
        for idx, crit in enumerate(ac_draft.criteria, 1):
            print(f"{idx}. {crit}")
            
        feedback = input("\nDo you confirm these AC, or would you like to modify them? (Type 'confirm' or provide edits):\n> ")
        
        if feedback.strip().lower() != 'confirm':
            update_user_prompt = self.prompts['user_prompts']['update_ac'].format(
                ac_json=ac_draft.model_dump_json(),
                feedback=feedback
            )
            final_ac = self.llm.query(sys_prompt, update_user_prompt, AcceptanceCriteria)
            self.ac = final_ac.criteria
        else:
            self.ac = ac_draft.criteria

    def _generate_final_story(self):
        print("\nGenerating final Jira artifact...")
        sys_prompt = self.prompts['system_personas']['story_writer']
        user_prompt = self.prompts['user_prompts']['generate_story'].format(
            state_json=self.state.model_dump_json(),
            agreed_ac=self.ac
        )
        
        final = self.llm.query(sys_prompt, user_prompt, FinalStory)
        
        print("\n" + "="*40)
        print("FINAL JIRA STORY")
        print("="*40)
        print(final.user_story)
        print("\nAcceptance Criteria:")
        for idx, crit in enumerate(final.acceptance_criteria, 1):
            print(f"{idx}. {crit}")
