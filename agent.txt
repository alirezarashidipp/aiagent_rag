import yaml
import asyncio
from typing import Tuple
from models import (
    SessionState, RawTextExtraction, ValidationResult, 
    MissingTechnicalDetail, AcceptanceCriteria, FinalStory
)
from llm_client import StructuredLLMClient

class JiraStatelessPipeline:
    def __init__(self, llm_client: StructuredLLMClient):
        self.llm = llm_client
        
        # In enterprise apps, inject this dependency instead of reading files in __init__
        with open("prompts.yaml", "r") as f:
            self.prompts = yaml.safe_load(f)
            
        self.personas = self.prompts["system_personas"]
        self.templates = self.prompts["user_prompts"]

    def run(self, raw_input: str) -> SessionState:
        """
        Synchronous wrapper. Bootstraps the asyncio event loop for CLI execution.
        """
        return asyncio.run(self._orchestrate_cli_loop(raw_input))

    async def _orchestrate_cli_loop(self, initial_input: str) -> SessionState:
        """
        Manages the interactive terminal loop and state persistence across turns.
        """
        state = SessionState()
        user_input = initial_input

        while True:
            reply, state = await self._execute_turn(user_input, state)
            print(f"\n[AGENT] {reply}")
            
            # Termination condition
            if getattr(state, "story_confirmed", False):
                break
                
            # Await terminal I/O (Blocking)
            user_input = input("\n[USER] > ").strip()
            if not user_input:
                user_input = "[INIT]"

        return state

    async def _execute_turn(self, user_input: str, state: SessionState) -> Tuple[str, SessionState]:
        """
        Pure state machine transition logic. No UI blocking here.
        """
        # Phase 0: Extraction
        if not state.what and not state.who and not state.why:
            prompt = self.templates["extract_raw_text"].format(raw_text=user_input)
            ext: RawTextExtraction = await self.llm.query(self.personas["extractor"], prompt, RawTextExtraction)
            
            state.what = ext.what.evidence if ext.what.identified else None
            state.who = ext.who.evidence if ext.who.identified else None
            state.why = ext.why.evidence if ext.why.identified else None
            state.ac_evidence = ext.ac_evidence
            
            return await self._execute_turn("[INIT]", state)

        # Phase 1: Validation
        missing = [f for f in ["what", "why", "who"] if getattr(state, f) is None]
        if missing:
            field = missing[0]
            if user_input == "[INIT]":
                return f"Missing field '{field.upper()}'. Please define it.", state
                
            prompt = self.templates["validate_missing"].format(field_name=field, user_ans=user_input)
            val: ValidationResult = await self.llm.query(self.personas["validator"], prompt, ValidationResult)
            
            if val.is_valid:
                setattr(state, field, val.normalized_value)
                return await self._execute_turn("[INIT]", state)
            return f"Rejected: {val.rejection_reason}. Try again.", state

        # Phase 2: Technical Grooming
        if getattr(state, "pending_questions", None) is None:
            prompt = self.templates["generate_dynamic_questions"].format(what_statement=state.what)
            tech: MissingTechnicalDetail = await self.llm.query(self.personas["tech_lead"], prompt, MissingTechnicalDetail)
            state.pending_questions = tech.questions
            return await self._execute_turn("[INIT]", state)

        if state.pending_questions:
            if user_input != "[INIT]":
                if user_input.strip().lower() != "skip":
                    state.tech_notes.append(f"Q: {state.pending_questions[0]} | A: {user_input}")
                state.pending_questions.pop(0)
                
            if state.pending_questions:
                return f"Tech Question: {state.pending_questions[0]}", state
            return await self._execute_turn("[INIT]", state)

        # Phase 3: Acceptance Criteria Synthesis
        if not state.ac_list:
            prompt = self.templates["generate_ac"].format(
                who=state.who, what=state.what, why=state.why,
                tech_notes="\n".join(state.tech_notes), ac_evidence=state.ac_evidence or "None"
            )
            ac: AcceptanceCriteria = await self.llm.query(self.personas["agile_coach"], prompt, AcceptanceCriteria)
            state.ac_list = ac.criteria
            return "Draft AC generated. Type 'confirm' to proceed or write feedback:\n" + "\n".join(state.ac_list), state

        if not getattr(state, "ac_confirmed", False):
            if user_input.strip().lower() == "confirm":
                state.ac_confirmed = True
                return await self._execute_turn("[INIT]", state)
                
            prompt = self.templates["update_ac"].format(ac_json=state.ac_list, feedback=user_input)
            ac_updated: AcceptanceCriteria = await self.llm.query(self.personas["agile_coach"], prompt, AcceptanceCriteria)
            state.ac_list = ac_updated.criteria
            return "AC Updated. Type 'confirm' to proceed or write feedback:\n" + "\n".join(state.ac_list), state

        # Phase 4: Final Story Generation
        if not state.final_story:
            prompt = self.templates["generate_story"].format(
                who=state.who, what=state.what, why=state.why, agreed_ac=state.ac_list
            )
            story: FinalStory = await self.llm.query(self.personas["story_writer"], prompt, FinalStory)
            state.final_story = story.user_story
            return f"Draft Story:\n{state.final_story}\n\nType 'confirm' to finalize or write feedback.", state

        if not getattr(state, "story_confirmed", False):
            if user_input.strip().lower() == "confirm":
                state.story_confirmed = True
                return "Process Complete.", state
                
            prompt = self.templates["update_story"].format(story_json=state.final_story, feedback=user_input)
            story_updated: FinalStory = await self.llm.query(self.personas["story_writer"], prompt, FinalStory)
            state.final_story = story_updated.user_story
            state.ac_list = story_updated.acceptance_criteria
            return f"Story Updated:\n{state.final_story}\n\nType 'confirm' to finalize.", state

        return "Pipeline execution finished.", state
