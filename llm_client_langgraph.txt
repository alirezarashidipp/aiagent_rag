import uuid
import ssl
import json
import httpx
import truststore
from openai import AsyncOpenAI, APIStatusError
from pydantic import BaseModel, ValidationError
from typing import Type, TypeVar
from token_jwt import get_token

T = TypeVar('T', bound=BaseModel)

class StructuredLLMClient:
    def __init__(self, base_url: str = "Link", user_id: str = "UC1939194", model_name: str = "GPT-4.1"):
        self.base_url = base_url
        self.user_id = user_id
        self.model_name = model_name
        
        ctx = truststore.SSLContext(ssl.PROTOCOL_TLS_CLIENT)
        
        self.httpx_client = httpx.AsyncClient(
            http2=True,
            verify=ctx,
            timeout=60.0
        )
        
        self.client = AsyncOpenAI(
            api_key="proxy-dummy-key",
            http_client=self.httpx_client,
            base_url=self.base_url
        )

    async def query(self, system_prompt: str, user_prompt: str, response_model: Type[T]) -> T:
        # Note: If get_token() performs network I/O, it must also be refactored to async.
        current_jwt = get_token()
        
        per_request_headers = {
            "X-HSBC-E2E-Trust-Token": current_jwt,
            "Content-Type": "application/json",
            "x-correlation-id": str(uuid.uuid4()),
            "x-usersession-id": str(uuid.uuid4())
        }

        schema_instruction = (
            f"\n\nYou MUST return a valid JSON object that adheres exactly to the following JSON schema:\n"
            f"{json.dumps(response_model.model_json_schema())}"
        )
        
        system_content = system_prompt + schema_instruction

        try:
            response = await self.client.chat.completions.create(
                model=self.model_name,
                messages=[
                    {"role": "system", "content": system_content},
                    {"role": "user", "content": user_prompt}
                ],
                user=self.user_id,
                extra_headers=per_request_headers,
                response_format={"type": "json_object"},
                max_tokens=1500,
                temperature=0.0
            )
            
            raw_content = response.choices[0].message.content
            return response_model.model_validate_json(raw_content)
            
        except APIStatusError as e:
            print(f"[FATAL] API Error: {e.status_code} - {e.response.text}")
            raise
        except ValidationError as e:
            print(f"[FATAL] LLM Output Validation Failed: {e}")
            raise
